```javascript
// Solution

function getSum(a, b) {

    let sum = 0;
    let smallerNumber = a;
    let largerNumber = b;

    if (a > b) {
        smallerNumber = b;
        largerNumber = a;
    }

    for (let i = smallerNumber; i <= largerNumber; i++) {
        sum += i;
    }
    return sum;
}

getSum(1, 0) == 1   // 1 + 0 = 1
getSum(1, 2) == 3   // 1 + 2 = 3
getSum(0, 1) == 1   // 0 + 1 = 1
getSum(1, 1) == 1   // 1 Since both are same
getSum(-1, 0) == -1 // -1 + 0 = -1
getSum(-1, 2) == 2  // -1 + 0 + 1 + 2 = 2
```
